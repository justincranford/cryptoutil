type JwkGenService struct {
	telemetryService      *cryptoutilTelemetry.TelemetryService
	rsa4096KeyGenPool     *cryptoutilPool.ValueGenPool[joseJwk.Key] // 512-bytes
	rsa3072KeyGenPool     *cryptoutilPool.ValueGenPool[joseJwk.Key] // 384-bytes
	rsa2048KeyGenPool     *cryptoutilPool.ValueGenPool[joseJwk.Key] // 256-bytes
	ecdsaP521KeyGenPool   *cryptoutilPool.ValueGenPool[joseJwk.Key] // 65.125-bytes
	ecdsaP384KeyGenPool   *cryptoutilPool.ValueGenPool[joseJwk.Key] // 48-bytes
	ecdsaP256KeyGenPool   *cryptoutilPool.ValueGenPool[joseJwk.Key] // 32-bytes
	ecdhP521KeyGenPool    *cryptoutilPool.ValueGenPool[joseJwk.Key] // 65.125-bytes
	ecdhP384KeyGenPool    *cryptoutilPool.ValueGenPool[joseJwk.Key] // 48-bytes
	ecdhP256KeyGenPool    *cryptoutilPool.ValueGenPool[joseJwk.Key] // 32-bytes
	ed25519KeyGenPool     *cryptoutilPool.ValueGenPool[joseJwk.Key] // 32-bytes
	aes256KeyGenPool      *cryptoutilPool.ValueGenPool[joseJwk.Key] // 32-bytes A256GCM, A256KW, A256GCMKW
	aes192KeyGenPool      *cryptoutilPool.ValueGenPool[joseJwk.Key] // 24-bytes A192GCM, A192KW, A192GCMKW
	aes128KeyGenPool      *cryptoutilPool.ValueGenPool[joseJwk.Key] // 16-bytes A128GCM, A128KW, A128GCMKW
	aes256HS512KeyGenPool *cryptoutilPool.ValueGenPool[joseJwk.Key] // 32-bytes A256CBC + 32-bytes HS512 (half of 64-bytes)
	aes192HS384KeyGenPool *cryptoutilPool.ValueGenPool[joseJwk.Key] // 24-bytes A192CBC + 24-bytes HS384 (half of 48-bytes)
	aes128HS256KeyGenPool *cryptoutilPool.ValueGenPool[joseJwk.Key] // 16-bytes A128CBC + 16-bytes HS256 (half of 32-bytes)
	hmac512KeyGenPool     *cryptoutilPool.ValueGenPool[joseJwk.Key] // 64-bytes HS512
	hmac384KeyGenPool     *cryptoutilPool.ValueGenPool[joseJwk.Key] // 48-bytes HS384
	hmac256KeyGenPool     *cryptoutilPool.ValueGenPool[joseJwk.Key] // 32-bytes HS256
	uuidV7KeyGenPool      *cryptoutilPool.ValueGenPool[*googleUuid.UUID]
}

	rsa4096KeyGenPool, err1 := cryptoutilPool.NewValueGenPool(cryptoutilPool.NewValueGenPoolConfig(ctx, telemetryService, "JwkGenService RSA 4096", 9, 9, cryptoutilPool.MaxLifetimeValues, cryptoutilPool.MaxLifetimeDuration, GenerateRSAJwkFunction(4096)))
	rsa3072KeyGenPool, err2 := cryptoutilPool.NewValueGenPool(cryptoutilPool.NewValueGenPoolConfig(ctx, telemetryService, "JwkGenService RSA 3072", 6, 6, cryptoutilPool.MaxLifetimeValues, cryptoutilPool.MaxLifetimeDuration, GenerateRSAJwkFunction(3072)))
	rsa2048KeyGenPool, err3 := cryptoutilPool.NewValueGenPool(cryptoutilPool.NewValueGenPoolConfig(ctx, telemetryService, "JwkGenService RSA 2048", 3, 3, cryptoutilPool.MaxLifetimeValues, cryptoutilPool.MaxLifetimeDuration, GenerateRSAJwkFunction(2048)))
	ecdsaP521KeyGenPool, err4 := cryptoutilPool.NewValueGenPool(cryptoutilPool.NewValueGenPoolConfig(ctx, telemetryService, "JwkGenService ECDSA-P521", 3, 9, cryptoutilPool.MaxLifetimeValues, cryptoutilPool.MaxLifetimeDuration, GenerateECDSAJwkFunction(elliptic.P521())))
	ecdsaP384KeyGenPool, err5 := cryptoutilPool.NewValueGenPool(cryptoutilPool.NewValueGenPoolConfig(ctx, telemetryService, "JwkGenService ECDSA-P384", 2, 6, cryptoutilPool.MaxLifetimeValues, cryptoutilPool.MaxLifetimeDuration, GenerateECDSAJwkFunction(elliptic.P384())))
	ecdsaP256KeyGenPool, err6 := cryptoutilPool.NewValueGenPool(cryptoutilPool.NewValueGenPoolConfig(ctx, telemetryService, "JwkGenService ECDSA-P256", 1, 3, cryptoutilPool.MaxLifetimeValues, cryptoutilPool.MaxLifetimeDuration, GenerateECDSAJwkFunction(elliptic.P256())))
	ecdhP521KeyGenPool, err7 := cryptoutilPool.NewValueGenPool(cryptoutilPool.NewValueGenPoolConfig(ctx, telemetryService, "JwkGenService ECDH-P521", 3, 9, cryptoutilPool.MaxLifetimeValues, cryptoutilPool.MaxLifetimeDuration, GenerateECDHJwkFunction(ecdh.P521())))
	ecdhP384KeyGenPool, err8 := cryptoutilPool.NewValueGenPool(cryptoutilPool.NewValueGenPoolConfig(ctx, telemetryService, "JwkGenService ECSH-P384", 2, 6, cryptoutilPool.MaxLifetimeValues, cryptoutilPool.MaxLifetimeDuration, GenerateECDHJwkFunction(ecdh.P384())))

	aes128KeyGenPool, err13 := cryptoutilPool.NewValueGenPool(cryptoutilPool.NewValueGenPoolConfig(ctx, telemetryService, "JwkGenService AES-128-GCM", 1, 3, cryptoutilPool.MaxLifetimeValues, cryptoutilPool.MaxLifetimeDuration, GenerateAESJwkFunction(128)))

	aes192HS384KeyGenPool, err15 := cryptoutilPool.NewValueGenPool(cryptoutilPool.NewValueGenPoolConfig(ctx, telemetryService, "JwkGenService AES-192-CBC HS-384", 2, 6, cryptoutilPool.MaxLifetimeValues, cryptoutilPool.MaxLifetimeDuration, GenerateAESHSJwkFunction(384)))

func (s *JwkGenService) GenerateJweJwk(enc *joseJwa.ContentEncryptionAlgorithm, alg *joseJwa.KeyEncryptionAlgorithm) (*googleUuid.UUID, joseJwk.Key, []byte, error) {
	var jwk joseJwk.Key
	switch *alg {
	case AlgDir:
		switch *enc {
		case EncA256GCM:
			jwk = s.aes256KeyGenPool.Get()
		case EncA192GCM:
			jwk = s.aes192KeyGenPool.Get()
		case EncA128GCM:
			jwk = s.aes128KeyGenPool.Get()
		case EncA256CBC_HS512:
			jwk = s.aes256HS512KeyGenPool.Get()
		case EncA192CBC_HS384:
			jwk = s.aes192HS384KeyGenPool.Get()
		case EncA128CBC_HS256:
			jwk = s.aes128HS256KeyGenPool.Get()
		default:
			return nil, nil, nil, fmt.Errorf("unsupported JWE JWK enc %s", *enc)
		}

	case AlgA256KW, AlgA256GCMKW:
		jwk = s.aes256KeyGenPool.Get()
	case AlgA192KW, AlgA192GCMKW:
		jwk = s.aes192KeyGenPool.Get()
	case AlgA128KW, AlgA128GCMKW:
		jwk = s.aes128KeyGenPool.Get()

	case AlgRSAOAEP512:
		jwk = s.rsa4096KeyGenPool.Get()
	case AlgRSAOAEP384:
		jwk = s.rsa3072KeyGenPool.Get()
	case AlgRSAOAEP256, AlgRSA15, AlgRSAOAEP:
		jwk = s.rsa2048KeyGenPool.Get()

	case AlgECDHES, AlgECDHESA256KW:
		jwk = s.ecdhP521KeyGenPool.Get()
	case AlgECDHESA192KW:
		jwk = s.ecdhP384KeyGenPool.Get()
	case AlgECDHESA128KW:
		jwk = s.ecdhP256KeyGenPool.Get()

	default:
		return nil, nil, nil, fmt.Errorf("unsupported JWE JWK alg %s", *alg)
	}

	kid, err := ExtractKidUuid(jwk)
	if err != nil {
		return nil, nil, nil, fmt.Errorf("failed to get `kid` header in JWE JWK: %w", err)
	}
	if err = jwk.Set(joseJwk.AlgorithmKey, *alg); err != nil {
		return nil, nil, nil, fmt.Errorf("failed to set `alg` header in JWE JWK: %w", err)
	}
	if err = jwk.Set("enc", *enc); err != nil {
		return nil, nil, nil, fmt.Errorf("failed to set `alg` header in JWE JWK: %w", err)
	}
	if err = jwk.Set(joseJwk.KeyUsageKey, "enc"); err != nil {
		return nil, nil, nil, fmt.Errorf("failed to set `use` header in JWE JWK: %w", err)
	}
	if err = jwk.Set(joseJwk.KeyOpsKey, OpsEncDec); err != nil {
		return nil, nil, nil, fmt.Errorf("failed to set `ops` header in JWK: %w", err)
	}
	encodedJwk, err := json.Marshal(jwk)
	if err != nil {
		return nil, nil, nil, fmt.Errorf("failed to serialize JWE JWK: %w", err)
	}
	return kid, jwk, encodedJwk, err
}

func (s *JwkGenService) GenerateJwsJwk(alg *joseJwa.SignatureAlgorithm) (*googleUuid.UUID, joseJwk.Key, []byte, error) {
	var jwk joseJwk.Key
	switch *alg {
	case AlgRS512, AlgPS512:
		jwk = s.rsa4096KeyGenPool.Get()
	case AlgRS384, AlgPS384:
		jwk = s.rsa3072KeyGenPool.Get()
	case AlgRS256, AlgPS256:
		jwk = s.rsa2048KeyGenPool.Get()
	case AlgES512:
		jwk = s.ecdsaP521KeyGenPool.Get()
	case AlgES384:
		jwk = s.ecdsaP384KeyGenPool.Get()
	case AlgES256:
		jwk = s.ecdsaP256KeyGenPool.Get()
	case AlgEdDSA:
		jwk = s.ed25519KeyGenPool.Get()
	case AlgHS512:
		jwk = s.hmac512KeyGenPool.Get()
	case AlgHS384:
		jwk = s.hmac384KeyGenPool.Get()
	case AlgHS256:
		jwk = s.hmac256KeyGenPool.Get()
	default:
		return nil, nil, nil, fmt.Errorf("unsupported JWS JWK alg: %s", alg)
	}

	kid, err := ExtractKidUuid(jwk)
	if err != nil {
		return nil, nil, nil, fmt.Errorf("failed to get `kid` header in JWE JWK: %w", err)
	}
	if err = jwk.Set(joseJwk.AlgorithmKey, *alg); err != nil {
		return nil, nil, nil, fmt.Errorf("failed to set `alg` header in JWE JWK: %w", err)
	}
	if err = jwk.Set(joseJwk.KeyUsageKey, "sign"); err != nil {
		return nil, nil, nil, fmt.Errorf("failed to set `use` header in JWE JWK: %w", err)
	}
	if err = jwk.Set(joseJwk.KeyOpsKey, OpsSigVer); err != nil {
		return nil, nil, nil, fmt.Errorf("failed to set `ops` header in JWK: %w", err)
	}
	encodedJwk, err := json.Marshal(jwk)
	if err != nil {
		return nil, nil, nil, fmt.Errorf("failed to serialize JWE JWK: %w", err)
	}
	return kid, jwk, encodedJwk, err
}
